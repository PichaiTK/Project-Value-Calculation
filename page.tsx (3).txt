function HealthItem({ name, status, latency, uptime }: any) {
  const statusColors: any = {
    nominal: "bg-emerald-500",
    degraded: "bg-amber-500",
    critical: "bg-red-500",
  }

  return (
    <div className="flex items-center justify-between p-4 rounded-xl bg-muted/30 border border-border/20 group hover:border-primary/30 transition-colors">
      <div className="space-y-1">
        <p className="text-sm font-semibold">{name}</p>
        <div className="flex items-center gap-2 text-[10px] text-muted-foreground font-mono">
          <span>LAT: {latency}</span>
          <span className="text-primary/50">|</span>
          <span>UP: {uptime}</span>
        </div>
      </div>
      <div
        className={`w-2 h-2 rounded-full ${statusColors[status]} animate-pulse shadow-[0_0_8px_rgba(16,185,129,0.5)]`}
      />
    </div>
  )
}

function LoadMetric({ label, value }: { label: string; value: number }) {
  const color = value > 85 ? "bg-red-500" : value > 70 ? "bg-amber-500" : "bg-primary"
  return (
    <div className="space-y-2">
      <div className="flex items-center justify-between text-sm">
        <span className="text-muted-foreground font-medium">{label}</span>
        <span className="font-mono font-bold">{value}%</span>
      </div>
      <div className="h-2 bg-muted/50 rounded-full overflow-hidden border border-border/20">
        <div className={`h-full ${color} transition-all duration-500 ease-out`} style={{ width: `${value}%` }} />
      </div>
    </div>
  )
}

function TrafficRegion({ name, value, active }: { name: string; value: string; active?: boolean }) {
  return (
    <div className="flex items-center justify-between p-3 rounded-lg bg-muted/20 border border-border/10">
      <span className={`text-sm ${active ? "text-foreground font-medium" : "text-muted-foreground"}`}>{name}</span>
      <span className="font-mono text-sm font-bold text-primary">{value}</span>
    </div>
  )
}
